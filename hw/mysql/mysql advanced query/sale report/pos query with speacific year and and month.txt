SELECT MONTH(ospos_sales.sale_time), YEAR(ospos_sales.sale_time), COALESCE(SUM(ospos_sales_items.quantity_purchased), 0) AS total_sum FROM `ospos_sales_items` JOIN ospos_sales on ospos_sales.sale_id = ospos_sales_items.sale_id where MONTH(ospos_sales.sale_time) = MONTH('2014-09-28') AND YEAR(ospos_sales.sale_time) = YEAR('2014-09-28') GROUP BY YEAR(ospos_sales.sale_time), MONTH(ospos_sales.sale_time)
